/* ============================================================
* BIT-LEVEL ZIPF SAMPLING CLASS
* ============================================================ */
public static class BitLevelSampler
{
    //  returns a list of (8) doubles where each value is a Zipf-distributed frequency
    //  the weights are then used to sample the actual vectors
    //  --> the logic:
    //  depending on the index position of the value in the vector: bit at position 0 is the most likely to be 1,
    //  bit at position 1 is less likely, bit at position n-1 (7 if there are 8 bits) is the least likely to be 1
    public static double[] ZipfWeights(int n, double s = 1.0)
    {
            double[] weights = new double[n];
            
            for (int i = 0; i < n; i++)
            {
                weights[i] = 1.0 / Math.Pow(i + 1, s);
            }

            /*
            Each item gets a weight inversely proportional to its rank raised to power s. 

            with s=0 ===> 1/i^0 = 1/1
            Rank 1: weight = 1/1 = 1
            Rank 2: weight = 1/1 = 1
            Rank 3: weight = 1/1 = 1
            Rank 10: weight = 1/1 = 1
            Each item has the same probability.

            With s=1:
            Rank 1: weight = 1/1 = 1
            Rank 2: weight = 1/2 = 0.5
            Rank 3: weight = 1/3 = 0.333
            Rank 10: weight = 1/10 = 0.1
            The first items get disproportionately high probabilities:
            With n=10, s=1.0, rank 1 gets ~29% probability, rank 2 gets ~15%, while rank 10 gets only ~3%

            With s=2:
            Rank 1: weight = 1/1 = 1
            Rank 2: weight = 1/4 = 0.25
            Rank 3: weight = 1/9 = 0.111
            Rank 10: weight = 1/100 = 0.01

            */

            // first bit has probability 0.5 of being 1
            // The follwing scale accordingly
            double firstWeight = weights[0];
            
            for (int i = 0; i < n; i++)
            {
                weights[i] = 0.5 * (weights[i] / firstWeight);
            }

            return weights;
    }


    // Returns a vector of meaning-signal pairs
    // by creating a meaning vector of length sampleSize (which normally is Bsize, and commonly has value 75) 
    // and each element is a vector of meaningSize (which is 8)
    // So, in the most common situation we get a 75-long list and each entry is a pair of two 8-long list
    // where each value is a double.
    public static List<(double[] meaning, double[] signal)> SampleBitLevelZipf(
    // Steps:
    // 1. It generates a Zipf distribution for 8 elements, meaning a vector of 8 elements where
    // the first is the probability of bit zero, the second of bit 1, and so on.
    // ---> This is the vector for s=1:
    //      Bit 0: 0.3679
    //      Bit 1: 0.1840
    //      Bit 2: 0.1226
    //      Bit 3: 0.0920
    //      Bit 4: 0.0736
    //      Bit 5: 0.0613
    //      Bit 6: 0.0526
    //      Bit 7: 0.0460
    // 2. it generates meaning vectors this way:
    // if a random number is below the threshold of the probability for the respective
    // position bit, then the value is 1, otherwise it's 0
    // 3. For each meaning, it calculates the corresponding signal using ForwardEncoder.

        Agent tutor,
        int sampleSize,
        int meaningSize,
        double s,
        double burstStopProbability,
        Random rng)
    {
        // Calculate Zipf probabilities for each bit position
        // Bit 0 gets highest probability, bit 7 gets lowest
        double[] bitProbabilities = ZipfWeights(meaningSize, s);

        var samples = new List<(double[], double[])>(sampleSize);
        for (int i = 0; i < sampleSize; i++)
        {
            // Generate a meaning where each bit is 1 with probability from Zipf
            double[] meaning = new double[meaningSize];
            for (int bit = 0; bit < meaningSize; bit++)
            {
                // if a random number is below the threshold of the probability for the respective
                // position bit, then the value is 1, otherwise it's 0
                meaning[bit] = rng.NextDouble() < bitProbabilities[bit] ? 1.0 : 0.0;
            }

            // Get the signal from tutor
            double[] signal = tutor.ForwardEncoder(meaning);
            double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            samples.Add((meaning, roundedSignal));
            
            while (rng.NextDouble() < burstStopProbability && samples.Count < sampleSize)
            {
                samples.Add((meaning, roundedSignal));
                i++;
            }
        }
        return samples;
    }
}

/* ============================================================
* 1. SENTENCE-LEVEL ZIPF SAMPLING CLASS
* ============================================================ */
public static class SentenceLevelSampler
{
    // Zipf weights over n ranks, normalized.
    public static double[] ZipfWeights(int n, double s = 1.0)
    {
        double[] weights = new double[n];
        double sum = 0.0;
        for (int i = 0; i < n; i++)
        {
            weights[i] = 1.0 / Math.Pow(i + 1, s);
            sum += weights[i];
        }
        for (int i = 0; i < n; i++)
            weights[i] /= sum;
        return weights;
    }

    /* ============================================================
     * 2. WEIGHTED INDEX SAMPLING (roulette wheel)
     * ============================================================ */
    private static int WeightedIndex(double[] weights, Random rng)
    {
        double total = 0.0;
        for (int i = 0; i < weights.Length; i++)
            total += weights[i];

        double r = rng.NextDouble() * total;

        double cumulative = 0.0;
        for (int i = 0; i < weights.Length; i++)
        {
            cumulative += weights[i];
            if (r < cumulative)
                return i;
        }
        return weights.Length - 1;
    }

    // Decode an integer rank-index into a meaning vector.
    // Convention: bit j of the index -> meaning[j] (same array convention as the bit-level sampler).
    private static double[] IndexToMeaning(int index, int meaningSize)
    {
        double[] meaning = new double[meaningSize];
        for (int j = 0; j < meaningSize; j++)
            meaning[j] = ((index >> j) & 1) == 1 ? 1.0 : 0.0;
        return meaning;
    }

    /* ============================================================
     * 3. SENTENCE-LEVEL ZIPF SAMPLER (drop-in for SampleBitLevelZipf)
     * ============================================================
     * - Zipf over WHOLE meanings (2^meaningSize "word types"), not per bit
     * - temporal clustering (bursts)
     * - sampling WITH replacement (human-like exposure)
     * ============================================================ */
    public static List<(double[] meaning, double[] signal)> SampleSentenceLevelZipf(
        Agent tutor,
        int sampleSize,
        int meaningSize,
        double s,
        double burstStopProbability,
        Random rng)
        {
            int vocabSize = 1 << meaningSize;             // 2^meaningSize (256 for 8 bits)
            double[] weights = ZipfWeights(vocabSize, s); // Zipf over the whole vocabulary

            var samples = new List<(double[], double[])>(sampleSize);

            for (int i = 0; i < sampleSize; i++)
            {
                // Draw one whole meaning from the Zipf distribution over all 2^meaningSize strings
                int index = WeightedIndex(weights, rng);
                double[] meaning = IndexToMeaning(index, meaningSize);

                double[] signal = tutor.ForwardEncoder(meaning);
                double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

                samples.Add((meaning, roundedSignal));

                // Burst: repeat the same meaning while still drawing under the stop probability
                while (rng.NextDouble() < burstStopProbability && samples.Count < sampleSize)
                {
                    samples.Add((meaning, roundedSignal));
                    i++;
                }
            }

            return samples;
        }
}

public static class GenerationalLearning // methods to learn language from one generation to the next
{
    private static System.Random StaticRandom = new System.Random();

    public static void IteratedLearning(int whole_evolution_repeats, int generations, int epochs, int meaningSize, int hiddenSize, int signalSize, int ASize, int BSize, double learningRate, float noise, int enc_loop, int dec_loop, int auto_loop, double s, double p)
    {

        for (int rep=0;rep<whole_evolution_repeats;rep++)
        {
            Console.WriteLine($"Repeat: {rep}/{whole_evolution_repeats}");
            StaticRandom = new Random(rep);
            Agent tutor = new Agent(meaningSize, hiddenSize, signalSize, 0, learningRate);
            var allMeanings = MeaningSpaceGenerator.GenerateMeaningSpace(meaningSize);

            for (int gen = 1; gen <= generations; gen++)
            {
                // pass learningRate to pupil as well
                Agent pupil = new Agent(meaningSize, hiddenSize, signalSize, gen, learningRate);
                trainAgent(tutor, pupil, ASize, BSize, allMeanings, meaningSize, epochs, learningRate, noise, gen, enc_loop, dec_loop, auto_loop, s, p);
                tutor = pupil;

                if (gen == generations)
                {
                    string path = $"language_evolved_after_{gen}_generations.csv";
                    var language = MeaningSpaceGenerator.GenerateSupervisedData(tutor, allMeanings, noise=0);
                    Csv.SaveCsv(path, language);
                }
            }
        }

    }

    var bitStats = Metrics.AnalyzeBitCompositionality(pupil, meanings, B1_analysis);
    float languageAdherence_final = (float)bitStats
        .Where(b => b.meaningBit >= 0)
        .Select(b => b.adherence)
        .DefaultIfEmpty(0)
        .Average();

public static float Stability(Agent tutor, Agent pupil, List<double[]> allMeanings)
{
    int matches = 0;
    
    for (int i = 0; i < allMeanings.Count; i++)
    {
        double[] meaning = allMeanings[i];
        double[] tutorSignal = tutor.ForwardEncoder(meaning);
        double[] pupilMeaning = pupil.ForwardDecoder(tutorSignal);
        double[] roundedPupilMeaning = pupilMeaning.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();
        if (meaning.SequenceEqual(roundedPupilMeaning)) matches++;

    }
    return (float)matches / allMeanings.Count;
}


public static float StabilityBitAccuracy(Agent tutor, Agent pupil, List<double[]> allMeanings)
{
    int totalBits = 0;
    int correctBits = 0;
    
    for (int i = 0; i < allMeanings.Count; i++)
    {
        double[] meaning = allMeanings[i];
        double[] tutorSignal = tutor.ForwardEncoder(meaning);
        double[] roundedTutorSignal = tutorSignal.Select(x => x > 0.5 ? 1.0 : 0.0).ToArray();
        
        double[] pupilMeaning = pupil.ForwardDecoder(roundedTutorSignal);
        double[] roundedPupilMeaning = pupilMeaning.Select(x => x > 0.5 ? 1.0 : 0.0).ToArray();
        
        for (int bit = 0; bit < meaning.Length; bit++)
        {
            if (meaning[bit] == roundedPupilMeaning[bit])
                correctBits++;
            totalBits++;
        }
    }
    
    return (float)correctBits / totalBits;
}


    public static float SignalBitAccuracy(Agent tutor, Agent pupil, List<double[]> allMeanings)
    {
        int totalBits = 0;
        int correctBits = 0;

        foreach (var meaning in allMeanings)
        {
            double[] tutorSignal = tutor.ForwardEncoder(meaning);
            double[] roundedTutorSignal = tutorSignal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            double[] pupilSignal = pupil.ForwardEncoder(meaning);
            double[] roundedPupilSignal = pupilSignal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            for (int bit = 0; bit < roundedTutorSignal.Length; bit++)
            {
                if (roundedTutorSignal[bit] == roundedPupilSignal[bit])
                    correctBits++;
                totalBits++;
            }
        }

        return (float)correctBits / totalBits;
    }

    // This measures how many distinct signals the agent produces across the whole meaning space.
    public static float Expressivity(Agent agent, List<double[]> allMeanings)
    {
        var signals = new List<double[]>();

        for (int i = 0; i < allMeanings.Count; i++)
        {
            double[] meaning = allMeanings[i];
            double[] signal = agent.ForwardEncoder(meaning);
            // Must round the signal here for metric calculation
            double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            signals.Add(roundedSignal);
        }
        int uniqueCount = signals.Select(v => string.Join(",", v)).Distinct().Count();
        float expressivity = (float)uniqueCount / allMeanings.Count;
        return expressivity;
    }




    // Eloise uses one-sided H(s_j | m_i = 1) (line 20: p ← sum(meaning[i,:] * signal[j,:]) / 2^(n-1), 
    // then entropy(p) — a single probability). Metrics.Compositionality uses two-sided 0.5·H(p1) + 0.5·H(p0). 
    // These coincide only for clean rules and diverge otherwise (the example from earlier: a bit that fires 
    // perfectly when m_i=1 but is random when m_i=0 scores 0 in the PDF, 0.5 in Metrics.Compositionality).
    public static float Compositionality_like_in_Eloise(Agent agent, List<double[]> allMeanings)
    {
        // The only edited lines are inside the signalCol loop: removed sum0, removed p0, and changed 
        // the assignment from 0.5 * CalculateEntropy(p1) + 0.5 * CalculateEntropy(p0) to CalculateEntropy(p1)
        int n = agent.autoencoder.hiddenSize; // assuming meaning size equals signal size equals n
        int numMessages = 1 << n; // 2^n
        int cnt = 0;

        // Initialise matrices
        int[,] meaningMatrix = new int[n, numMessages];
        int[,] signalMatrix = new int[n, numMessages];

        foreach (var m in allMeanings)
        {
            double[] signal = agent.ForwardEncoder(m);
            double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            for (int row = 0; row < n; row++)
            {
                meaningMatrix[row, cnt] = (int)m[row];
                signalMatrix[row, cnt] = (int)roundedSignal[row];
            }
            cnt++;
        }

        List<List<double>> entropy = new List<List<double>>();
        for (int i = 0; i < n; i++)
            entropy.Add(new List<double>());

        for (int mCol = 0; mCol < n; mCol++)
        {
            double[] currColEntropy = Enumerable.Repeat(1.0, n).ToArray();

            for (int signalCol = 0; signalCol < n; signalCol++)
            {
                double sum1 = 0;
                for (int msg = 0; msg < numMessages; msg++)
                {
                    sum1 += meaningMatrix[mCol, msg] * signalMatrix[signalCol, msg];
                }

                double p1 = sum1 / Math.Pow(2, n - 1);
                currColEntropy[signalCol] = CalculateEntropy(p1); // one-sided H(s_j | m_i=1), per PDF
            }

            int minIndex = Array.IndexOf(currColEntropy, currColEntropy.Min());
            entropy[minIndex].Add(currColEntropy[minIndex]);
        }

        double entropySum = 0;
        foreach (var vals in entropy)
        {
            entropySum += vals.Count > 0 ? vals.Min() : 1.0;
        }

        return 1f - (float)(entropySum / n);
    }


    public static float Compositionality(Agent agent, List<double[]> allMeanings)
    {
        int n = agent.autoencoder.hiddenSize; // assuming meaning size equals signal size equals n
        int numMessages = 1 << n; // 2^n
        int cnt = 0;

        // Initialise matrices
        int[,] meaningMatrix = new int[n, numMessages];
        int[,] signalMatrix = new int[n, numMessages];

        foreach (var m in allMeanings)
        {
            double[] signal = agent.ForwardEncoder(m);
            double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            for (int row = 0; row < n; row++)
            {
                meaningMatrix[row, cnt] = (int)m[row];
                signalMatrix[row, cnt] = (int)roundedSignal[row];
            }
            cnt++;
        }

        List<List<double>> entropy = new List<List<double>>();
        for (int i = 0; i < n; i++)
            entropy.Add(new List<double>());

        for (int mCol = 0; mCol < n; mCol++)
        {
            double[] currColEntropy = Enumerable.Repeat(1.0, n).ToArray();

            for (int signalCol = 0; signalCol < n; signalCol++)
            {
                double sum1 = 0;
                double sum0 = 0;
                for (int msg = 0; msg < numMessages; msg++)
                {
                    sum1 += meaningMatrix[mCol, msg] * signalMatrix[signalCol, msg];
                    sum0 += (1 - meaningMatrix[mCol, msg]) * signalMatrix[signalCol, msg];
                }

                double p1 = sum1 / Math.Pow(2, n - 1);
                double p0 = sum0 / Math.Pow(2, n - 1);
                currColEntropy[signalCol] = 0.5 * CalculateEntropy(p1) + 0.5 * CalculateEntropy(p0);
            }

            int minIndex = Array.IndexOf(currColEntropy, currColEntropy.Min());
            entropy[minIndex].Add(currColEntropy[minIndex]);
        }

        double entropySum = 0;
        foreach (var vals in entropy)
        {
            entropySum += vals.Count > 0 ? vals.Min() : 1.0;
        }

        return 1f - (float)(entropySum / n);
    }



public static List<BitStatistic> AnalyzeBitCompositionality(Agent agent, List<double[]> allMeanings, List<(double[] meaning, double[] signal)> B1)
{
    int n = agent.autoencoder.hiddenSize;  // 8
    int numMessages = (int)Math.Pow(2, n); // numMessages = max total number of combinations = 2^8 = 256
    int cnt = 0;

    // Initialize matrices for FULL meaning space (for compositionality calculation)
    int[,] meaningMatrix = new int[n, numMessages];
    int[,] signalMatrix = new int[n, numMessages];

    foreach (var m in allMeanings)
    {
        double[] signal = agent.ForwardEncoder(m);
        double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

        for (int row = 0; row < n; row++)
        {
            meaningMatrix[row, cnt] = (int)m[row];
            signalMatrix[row, cnt] = (int)roundedSignal[row];
        }
        cnt++;
    }

    // Calculate bit frequencies from FULL MEANING SPACE
    int[] meaningFreq_full = new int[n];
    int[] signalFreq_full = new int[n];

    for (int msg = 0; msg < numMessages; msg++) //0-255
    {
        for (int bit = 0; bit < n; bit++)
        {
            meaningFreq_full[bit] += meaningMatrix[bit, msg];
            signalFreq_full[bit] += signalMatrix[bit, msg];
        }
    }

    // Calculate bit frequencies from B1 (TRAINING DATA)
    int[] meaningFreq_B1 = new int[n];
    int[] signalFreq_B1 = new int[n];

    foreach (var (meaning, signal) in B1) //loop on the elements in B1
    {
        for (int bit = 0; bit < n; bit++)
        {
            meaningFreq_B1[bit] += (int)meaning[bit];
            signalFreq_B1[bit] += (int)signal[bit];
            //if (meaning[bit] >= 0.5) meaningFreq_B1[bit]++;
            //if (signal[bit] >= 0.5) signalFreq_B1[bit]++;
        }
    }

    // Display meaning bit frequencies from B1 (what the agent actually saw)
    Console.WriteLine("\n--- MEANING BIT FREQUENCIES IN B1 (training data) ---");
    for (int bit = 0; bit < n; bit++)
    {
        double freq = meaningFreq_B1[bit] / (double)B1.Count;
        Console.WriteLine($"Meaning bit {bit}: {meaningFreq_B1[bit]}/{B1.Count} = {freq:F4} ({freq*100:F2}%)");
    }
    Console.WriteLine();

    Console.WriteLine("--- SIGNAL BIT FREQUENCIES IN B1 (training data) ---");
    for (int bit = 0; bit < n; bit++)
    {
        double freq = signalFreq_B1[bit] / (double)B1.Count;
        Console.WriteLine($"Signal bit {bit}: {signalFreq_B1[bit]}/{B1.Count} = {freq:F4} ({freq*100:F2}%)");
    }
    Console.WriteLine();



    // Build full n×n table of entropy and pValue for every (meaningBit, signalBit) pair
    // This builds a complete n×n lookup table that answers, for every possible pairing of a meaning bit and a signal bit, 
    // two questions: how strongly are they related, and in which direction.
    double[,] allEntropies = new double[n, n];
    double[,] allPValues   = new double[n, n];

    for (int mCol = 0; mCol < n; mCol++)
    {
        for (int signalCol = 0; signalCol < n; signalCol++)
        {
            double sum1 = 0; // count(meaning=1 AND signal=1)
            double sum0 = 0; // count(meaning=0 AND signal=1)
            for (int msg = 0; msg < numMessages; msg++)
            {
                sum1 += meaningMatrix[mCol, msg] * signalMatrix[signalCol, msg];
                sum0 += (1 - meaningMatrix[mCol, msg]) * signalMatrix[signalCol, msg];
            }

            // P(signal=1 | meaning=1) and P(signal=1 | meaning=0)
            double p1 = sum1 / Math.Pow(2, n - 1);
            double p0 = sum0 / Math.Pow(2, n - 1);

            // True conditional entropy: average over both halves
            // Zero only when signal is perfectly determined in both directions
            double ent = 0.5 * CalculateEntropy(p1) + 0.5 * CalculateEntropy(p0);

            // pValue still based on p1 for correlation direction
            allPValues[mCol, signalCol]   = p1;
            allEntropies[mCol, signalCol] = ent;
        }
    }

    // Greedy one-to-one assignment: repeatedly pick the globally best remaining pair
    // so every signal bit gets exactly one meaning bit and vice versa
    int[]    signalToMeaning  = Enumerable.Repeat(-1,  n).ToArray();
    double[] assignedEntropy  = Enumerable.Repeat(1.0, n).ToArray();
    double[] assignedPValue   = Enumerable.Repeat(0.5, n).ToArray();

    bool[] meaningUsed = new bool[n];
    bool[] signalUsed  = new bool[n];

    for (int assignment = 0; assignment < n; assignment++)
    {
        double bestEnt = double.MaxValue;
        int bestM = -1, bestS = -1;

        for (int m = 0; m < n; m++)
        {
            if (meaningUsed[m]) continue;
            for (int s = 0; s < n; s++)
            {
                if (signalUsed[s]) continue;
                if (allEntropies[m, s] < bestEnt)
                {
                    bestEnt = allEntropies[m, s];
                    bestM = m;
                    bestS = s;
                }
            }
        }

        signalToMeaning[bestS] = bestM;
        assignedEntropy[bestS] = bestEnt;
        assignedPValue[bestS]  = allPValues[bestM, bestS];
        meaningUsed[bestM] = true;
        signalUsed[bestS]  = true;
    }

    // Collect statistics for each signal bit
    List<BitStatistic> bitStats = new List<BitStatistic>();

    for (int sBit = 0; sBit < n; sBit++)
    {
        int    mappedMeaningBit = signalToMeaning[sBit];
        double minEnt           = assignedEntropy[sBit];

        // convert the probability into a correlation-like value between -1 and +1
        // assignedPValue[sBit] is just P(signal bit = 1 | meaning bit = 1), i.e. a single probability between 0 and 1
        // for the specific (signal bit, meaning bit) pair that was chosen by the greedy assignment as the best match.
        double correlation      = 2.0 * assignedPValue[sBit] - 1.0;
        
        // USE B1 FREQUENCY HERE (not full space)
        double freq_B1 = signalFreq_B1[sBit] / (double)B1.Count;
        double freq_full = signalFreq_full[sBit] / (double)numMessages;

        // Per-bit adherence: over full meaning space, how often does this signal bit
        // match its mapped meaning bit (accounting for inversion)?
        double adherence = 0.0;
        if (mappedMeaningBit >= 0)
        {
            int adherenceCorrect = 0;
            foreach (var m in allMeanings)
            {
                double[] sig = agent.ForwardEncoder(m);
                double roundedSig = sig[sBit] >= 0.5 ? 1.0 : 0.0;

                // If the correlation between this signal bit and its mapped meaning bit is positive, 
                // the expected signal value is whatever the meaning bit is. 
                // If the correlation is negative, the expected signal value is the opposite of whatever the meaning bit is.
                double expected = correlation >= 0 ? m[mappedMeaningBit] : 1.0 - m[mappedMeaningBit];
                if (roundedSig == expected) adherenceCorrect++;
            }
            adherence = (double)adherenceCorrect / allMeanings.Count;
        }

        bitStats.Add(new BitStatistic
        {
            signalBit = sBit,
            entropy = minEnt,
            freq = freq_B1,
            freq_full = freq_full,
            meaningBit = mappedMeaningBit,
            correlation = correlation,
            adherence = adherence
        });
    }

    // Sort by compositionality
    bitStats.Sort((a, b) => a.entropy.CompareTo(b.entropy));

    Console.WriteLine("\n=== BIT-LEVEL COMPOSITIONALITY ANALYSIS ===");
    Console.WriteLine("Sorted by compositionality (most compositional first):");
    Console.WriteLine("SignalBit | Entropy | Freq(B1) | Freq(Full) | MeaningBit | Corr");
    Console.WriteLine("----------|---------|----------|------------|------------|-------");
    
    foreach (BitStatistic stat in bitStats)
    {
        string meaningStr = stat.meaningBit >= 0 ? stat.meaningBit.ToString() : "-";
        Console.WriteLine($"   {stat.signalBit,2}     | {stat.entropy:F4}  | {stat.freq:F4}   |  {stat.freq_full:F4}    |     {meaningStr,2}     | {stat.correlation,6:F3}");
    }

    // Calculate correlations using B1 frequencies
    double[] entropies = new double[bitStats.Count];
    double[] frequencies = new double[bitStats.Count];
    for (int i = 0; i < bitStats.Count; i++)
    {
        entropies[i] = bitStats[i].entropy;
        frequencies[i] = bitStats[i].freq;  // B1 frequency
    }

    double corrEntropyFreq = CalculateCorrelation(entropies, frequencies);

    Console.WriteLine($"\nCorrelation (Entropy vs B1 Frequency): {corrEntropyFreq:F4}");
    Console.WriteLine("  (negative correlation = more frequent bits in training are more compositional)\n");
    
    return bitStats;
}

    private static double CalculateCorrelation(double[] x, double[] y)
    {
        if (x.Length != y.Length || x.Length == 0)
            return 0;

        double meanX = x.Average();
        double meanY = y.Average();

        double numerator = 0;
        double sumXSquared = 0;
        double sumYSquared = 0;

        for (int i = 0; i < x.Length; i++)
        {
            double dx = x[i] - meanX;
            double dy = y[i] - meanY;
            numerator += dx * dy;
            sumXSquared += dx * dx;
            sumYSquared += dy * dy;
        }

        if (sumXSquared == 0 || sumYSquared == 0)
            return 0;

        return numerator / Math.Sqrt(sumXSquared * sumYSquared);
    }

    private static double CalculateEntropy(double p)
    {
        if (p == 0 || p == 1)
            return 0;
        return -p * Math.Log(p) / Math.Log(2) - (1 - p) * Math.Log(1 - p) / Math.Log(2);
    }

    
    public class BitStatistic
    {
        public int signalBit;
        public double entropy;
        public double freq;        // Frequency in B1 (training data)
        public double freq_full;   // Frequency in full meaning space
        public int meaningBit;
        public double correlation = 0.0;  // Actual correlation value
        public double adherence = 0.0;    // How often this signal bit follows its own mapping
    }

    // For each meaning in the full space, check how many of the pupil's signal bits
    // match the compositional ideal derived from bitStats.
    // The expected value of signal bit s is meaning[meaningBit] if correlation >= 0,
    // or 1 - meaning[meaningBit] if inverted (correlation < 0).
    // Bits with no valid mapping (meaningBit == -1) are skipped.
    public static float LanguageAdherence(Agent pupil, List<double[]> allMeanings, List<BitStatistic> bitStats)
    {
        // Filter to only bits that have a valid mapping
        var validMappings = bitStats.Where(b => b.meaningBit >= 0).ToList();
        if (validMappings.Count == 0) return 0f;

        int totalBits = 0;
        int correctBits = 0;

        foreach (var meaning in allMeanings)
        {
            double[] signal = pupil.ForwardEncoder(meaning);
            double[] roundedSignal = signal.Select(x => x >= 0.5 ? 1.0 : 0.0).ToArray();

            foreach (var map in validMappings)
            {
                double expected = map.correlation >= 0
                    ? meaning[map.meaningBit]
                    : 1.0 - meaning[map.meaningBit];

                if (roundedSignal[map.signalBit] == expected)
                    correctBits++;
                totalBits++;
            }
        }

        return (float)correctBits / totalBits;
    }
}

public static class PairUtils
{
    // Accept and return List<(double[] meaning, double[] signal)>
    public static List<(double[] meaning, double[] signal)> RandomSubset(
        List<(double[] meaning, double[] signal)> supervisedPairs,
        int subsetSize)
    {
        if (subsetSize <= 0 || subsetSize > supervisedPairs.Count)
            throw new ArgumentException("subsetSize must be between 1 and supervisedPairs.Count");

        System.Random rand = new System.Random();
        return supervisedPairs
            .OrderBy(x => rand.Next())
            .Take(subsetSize)
            .ToList();
    }

    public static List<(double[] meaning, double[] signal)> CopyAndShuffle(
        List<(double[] meaning, double[] signal)> subset)
    {
        var copy = new List<(double[] meaning, double[] signal)>(subset.Count);
        foreach (var (meaning, signal) in subset)
        {
            double[] newMeaning = new double[meaning.Length];
            double[] newSignal = new double[signal.Length];
            Array.Copy(meaning, newMeaning, meaning.Length);
            Array.Copy(signal, newSignal, signal.Length);
            copy.Add((newMeaning, newSignal));
        }

        System.Random rand = new System.Random();
        for (int i = copy.Count - 1; i > 0; i--)
        {
            int j = rand.Next(i + 1);
            (copy[i], copy[j]) = (copy[j], copy[i]);
        }

        return copy;
    }
}

public static class MeaningSpaceGenerator
{
    public static List<double[]> GenerateMeaningSpace(int bitN)
    {
        List<double[]> space = new List<double[]>();
        int total = (int)Math.Pow(2, bitN);

        for (int i = 0; i < total; i++)
        {
            space.Add(IntToBin(bitN, i));
        }

        return space;
    }

    private static double[] IntToBin(int bitN, int value)
    {
        double[] bits = new double[bitN];
        for (int j = 0; j < bitN; j++)
        {
            bits[bitN - j - 1] = (value >> j) & 1;
        }
        return bits;
    }

    public static List<(double[] meaning, double[] signal)> GenerateSupervisedData(Agent agent, List<double[]> meaningSpace, float noise)
    {
        var supervisedPairs = new List<(double[] meaning, double[] signal)>();
        System.Random rand = new System.Random();

        foreach (var meaning in meaningSpace)
        {
            //Console.WriteLine($"Noise: {noise}");
            double[] signal = agent.ForwardEncoder(meaning); //output is continuous
            
            // add noise
            for (int i = 0; i < signal.Length; i++)
                signal[i] += (rand.NextDouble() * 2 - 1) * (noise / 2); // add noise

            // round to get a binary signal
            double[] roundedSignal = Binary(signal);
            supervisedPairs.Add((meaning, roundedSignal));
        }

        return supervisedPairs;
    }

    public static double[] Binary(double[] input)
    {
        double[] binary = new double[input.Length];
        for (int i = 0; i < input.Length; i++)
            binary[i] = input[i] >= 0.5 ? 1.0 : 0.0;
        return binary;
    }
}
// ---------------------------------------------------------------------

public class MLP // Multi layer perceptron
{
    public int meaningSize, hiddenSize, signalSize;

    public double[,] We1;    // weights encoder layer 1->2 (meaningSize x hiddenSize)
    public double[,] We2;    // weights encoder layer 2->3 (hiddenSize x signalSize)
    public double[] Be1;     // biases encoder hidden layer
    public double[] Be2;     // biases encoder output layer

    public double[,] Wd1;    // weights decoder layer 1->2 (signalSize x hiddenSize)
    public double[,] Wd2;    // weights decoder layer 2->3 (hiddenSize x meaningSize)
    public double[] Bd1;     // biases decoder hidden layer
    public double[] Bd2;     // biases decoder output layer

    public double lr = 0.1; // learning rate

    public MLP(int input, int hidden, int output, double learningRate)
    {
        meaningSize = input;
        hiddenSize = hidden;
        signalSize = output;
        lr = learningRate;

        // Weight matrix dimensions
        We1 = new double[meaningSize, hiddenSize];
        We2 = new double[hiddenSize, signalSize];
        
        Wd1 = new double[signalSize, hiddenSize];     // Signal is the input to the decoder
        Wd2 = new double[hiddenSize, meaningSize];    // Meaning is the output of the decoder

        // Bias vector dimensions
        Be1 = new double[hiddenSize];
        Be2 = new double[signalSize];
        Bd1 = new double[hiddenSize];
        Bd2 = new double[meaningSize]; // Corrected size

        // INITIALIZATION OF THE WEIGHTS (Xavier/Glorot Uniform)
        double limit1 = Math.Sqrt(6.0 / (meaningSize + hiddenSize));
        double limit2 = Math.Sqrt(6.0 / (hiddenSize + signalSize)); // Using signalSize for output layer size calculation

        System.Random rnd = new System.Random();
        
        // Encoder weights (We1: meaning -> hidden)
        for (int i = 0; i < meaningSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                We1[i, j] = rnd.NextDouble() * 2 * limit1 - limit1;

        // Encoder weights (We2: hidden -> signal)
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < signalSize; j++)
                We2[i, j] = rnd.NextDouble() * 2 * limit2 - limit2;

        // Decoder weights (Wd1: signal -> hidden)
        for (int i = 0; i < signalSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                Wd1[i, j] = rnd.NextDouble() * 2 * limit1 - limit1; // Same bounds as We1

        // Decoder weights (Wd2: hidden -> meaning)
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < meaningSize; j++)
                Wd2[i, j] = rnd.NextDouble() * 2 * limit2 - limit2; // Same bounds as We2

        // FIX B: Initialize all biases to zero
        for (int i = 0; i < hiddenSize; i++)
        {
            Be1[i] = 0.0;
            Bd1[i] = 0.0;
        }

        for (int i = 0; i < signalSize; i++)
        {
            Be2[i] = 0.0;
        }

        for (int i = 0; i < meaningSize; i++)
        {
            Bd2[i] = 0.0;
        }
    }

    double Sigmoid(double x) => 1.0 / (1.0 + Math.Exp(-x));
    double SigmoidDerivative(double y) => y * (1.0 - y);

    // Mean squared error loss
    double MSE(double[] target, double[] output)
    {
        double loss = 0.0;
        for (int i = 0; i < output.Length; i++)
            loss += Math.Pow(target[i] - output[i], 2);
        return loss / output.Length;
    }

    public double[] ForwardEncoder(double[] input)
    {
        double[] hidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Be1[i];
            for (int j = 0; j < meaningSize; j++)
                sum += input[j] * We1[j, i];
            hidden[i] = Sigmoid(sum);
        }

        double[] output = new double[signalSize];
        for (int i = 0; i < signalSize; i++)
        {
            double sum = Be2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += hidden[j] * We2[j, i];
            output[i] = Sigmoid(sum);
        }
        return output;
    }


    public double[] ForwardDecoder(double[] input)
    {
        double[] hidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Bd1[i];
            for (int j = 0; j < signalSize; j++)
                sum += input[j] * Wd1[j, i];
            hidden[i] = Sigmoid(sum);
        }

        double[] output = new double[meaningSize];
        for (int i = 0; i < meaningSize; i++)
        {
            double sum = Bd2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += hidden[j] * Wd2[j, i];
            output[i] = Sigmoid(sum);
        }
        return output;
    }

    // ---------- TrainEncoder ----------
    public double TrainEncoder(double[] input, double[] target)
    {
        // FORWARD
        double[] hidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Be1[i];
            for (int j = 0; j < meaningSize; j++)
                sum += input[j] * We1[j, i];
            hidden[i] = Sigmoid(sum);
        }

        double[] output = new double[signalSize];
        for (int i = 0; i < signalSize; i++)
        {
            double sum = Be2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += hidden[j] * We2[j, i];
            output[i] = Sigmoid(sum);
        }

        double loss = MSE(target, output);

        // BACKPROP (MSE): delta_out = (output - target) * sigmoid'(output)
        double[] deltaOut = new double[signalSize];
        for (int i = 0; i < signalSize; i++)
            deltaOut[i] = (output[i] - target[i]) * SigmoidDerivative(output[i]);

        double[] deltaHidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = 0;
            for (int j = 0; j < signalSize; j++)
                sum += deltaOut[j] * We2[i, j];
            deltaHidden[i] = sum * SigmoidDerivative(hidden[i]);
        }

        // UPDATE We2, Be2
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < signalSize; j++)
                We2[i, j] -= lr * deltaOut[j] * hidden[i];

        for (int j = 0; j < signalSize; j++)
            Be2[j] -= lr * deltaOut[j];

        // UPDATE We1, Be1
        for (int i = 0; i < meaningSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                We1[i, j] -= lr * deltaHidden[j] * input[i];

        for (int j = 0; j < hiddenSize; j++)
            Be1[j] -= lr * deltaHidden[j];

        return loss;
    }

    // ---------- TrainDecoder ----------
    public double TrainDecoder(double[] input, double[] target)
    {
        // FORWARD
        double[] hidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Bd1[i];
            for (int j = 0; j < signalSize; j++)
                sum += input[j] * Wd1[j, i];
            hidden[i] = Sigmoid(sum);
        }

        double[] output = new double[meaningSize];
        for (int i = 0; i < meaningSize; i++)
        {
            double sum = Bd2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += hidden[j] * Wd2[j, i];
            output[i] = Sigmoid(sum);
        }

        double loss = MSE(target, output);

        // BACKPROP (MSE)
        double[] deltaOut = new double[meaningSize];
        for (int i = 0; i < meaningSize; i++)
            deltaOut[i] = (output[i] - target[i]) * SigmoidDerivative(output[i]);

        double[] deltaHidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = 0;
            for (int j = 0; j < meaningSize; j++)
                sum += deltaOut[j] * Wd2[i, j];
            deltaHidden[i] = sum * SigmoidDerivative(hidden[i]);
        }

        // UPDATE Wd2, Bd2
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < meaningSize; j++)
                Wd2[i, j] -= lr * deltaOut[j] * hidden[i];

        for (int j = 0; j < meaningSize; j++)
            Bd2[j] -= lr * deltaOut[j];

        // UPDATE Wd1, Bd1
        for (int i = 0; i < signalSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                Wd1[i, j] -= lr * deltaHidden[j] * input[i];

        for (int j = 0; j < hiddenSize; j++)
            Bd1[j] -= lr * deltaHidden[j];

        return loss;
    }

    // ---------- TrainAutoencoder ----------
    public double TrainAutoencoder(double[] input)
    {
        // ENCODER FORWARD
        double[] ehidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Be1[i];
            for (int j = 0; j < meaningSize; j++)
                sum += input[j] * We1[j, i];
            ehidden[i] = Sigmoid(sum);
        }

        double[] eoutput = new double[signalSize];
        for (int i = 0; i < signalSize; i++)
        {
            double sum = Be2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += ehidden[j] * We2[j, i];
            eoutput[i] = Sigmoid(sum);
        }

        // DECODER FORWARD
        double[] dhidden = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = Bd1[i];
            for (int j = 0; j < signalSize; j++)
                sum += eoutput[j] * Wd1[j, i];
            dhidden[i] = Sigmoid(sum);
        }

        double[] doutput = new double[meaningSize];
        for (int i = 0; i < meaningSize; i++)
        {
            double sum = Bd2[i];
            for (int j = 0; j < hiddenSize; j++)
                sum += dhidden[j] * Wd2[j, i];
            doutput[i] = Sigmoid(sum);
        }

        double[] target = input;
        double loss = MSE(target, doutput);

        // BACKPROP DECODER (MSE)
        double[] deltaOut = new double[meaningSize];
        for (int i = 0; i < meaningSize; i++)
            deltaOut[i] = (doutput[i] - target[i]) * SigmoidDerivative(doutput[i]);

        double[] deltaHiddenD = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = 0;
            for (int j = 0; j < meaningSize; j++)
                sum += deltaOut[j] * Wd2[i, j];
            deltaHiddenD[i] = sum * SigmoidDerivative(dhidden[i]);
        }

        // UPDATE DECODER weights (gradient descent)
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < meaningSize; j++)
                Wd2[i, j] -= lr * deltaOut[j] * dhidden[i];

        for (int j = 0; j < meaningSize; j++)
            Bd2[j] -= lr * deltaOut[j];

        for (int i = 0; i < signalSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                Wd1[i, j] -= lr * deltaHiddenD[j] * eoutput[i];

        for (int j = 0; j < hiddenSize; j++)
            Bd1[j] -= lr * deltaHiddenD[j];

        // BACKPROP TO ENCODER (treat decoder's upstream gradients as encoder output gradients)
        double[] deltaEoutput = new double[signalSize];
        for (int i = 0; i < signalSize; i++)
        {
            double sum = 0;
            for (int j = 0; j < hiddenSize; j++)
                sum += deltaHiddenD[j] * Wd1[i, j];
            deltaEoutput[i] = sum * SigmoidDerivative(eoutput[i]);
        }

        double[] deltaHiddenE = new double[hiddenSize];
        for (int i = 0; i < hiddenSize; i++)
        {
            double sum = 0;
            for (int j = 0; j < signalSize; j++)
                sum += deltaEoutput[j] * We2[i, j];
            deltaHiddenE[i] = sum * SigmoidDerivative(ehidden[i]);
        }

        // UPDATE ENCODER weights
        for (int i = 0; i < hiddenSize; i++)
            for (int j = 0; j < signalSize; j++)
                We2[i, j] -= lr * deltaEoutput[j] * ehidden[i]; // careful indexing: We2[hidden,signal]

        for (int i = 0; i < meaningSize; i++)
            for (int j = 0; j < hiddenSize; j++)
                We1[i, j] -= lr * deltaHiddenE[j] * input[i];

        for (int i = 0; i < signalSize; i++) Be2[i] -= lr * deltaEoutput[i];
        for (int i = 0; i < hiddenSize; i++) Be1[i] -= lr * deltaHiddenE[i];

        return loss;
    }


    private double BCE(double[] target, double[] output)
    {
        double loss = 0.0;
        for (int i = 0; i < output.Length; i++)
        {
            double p = Math.Max(1e-7, Math.Min(1 - 1e-7, output[i])); // Clip to avoid log(0)
            loss += -(target[i] * Math.Log(p) + (1 - target[i]) * Math.Log(1 - p));
        }
        return loss / output.Length;
    }




}

public static class MeaningSignalGenerator
{
    // Generates binary (signal, meaning) pairs for decoder testing
    // Each signal is a deterministic permutation of meaning bits
    public static List<(double[] signal, double[] meaning)> GenerateBinaryPairs(
        int nSamples, int meaningSize, int signalSize)
    {
        if (meaningSize != signalSize)
            throw new ArgumentException("For binary test pairs, meaningSize must equal signalSize.");

        var pairs = new List<(double[], double[])>();
        System.Random rand = new System.Random(42);

        for (int n = 0; n < nSamples; n++)
        {
            double[] meaning = new double[meaningSize];
            double[] signal = new double[signalSize];

            // generate random binary meaning vector
            for (int i = 0; i < meaningSize; i++)
                meaning[i] = rand.NextDouble() > 0.5 ? 1.0 : 0.0;

            // create a deterministic signal (e.g., rotate bits)
            for (int i = 0; i < signalSize; i++)
                signal[i] = meaning[(i + 2) % meaningSize]; // simple rotation by 2

            pairs.Add((signal, meaning));
        }

        return pairs;
    }
}



public static class Csv
{
    public static void SaveCsv(string path, List<(double[] meaning, double[] signal)> list)
    {
        using var writer = new StreamWriter(path);

        foreach (var (meaning, signal) in list)
        {
            var row = string.Join(",", meaning.Concat(signal));
            writer.WriteLine(row);
        }
    }


    public static List<(double[] meaning, double[] signal)> LoadCsv(string path, int meaningLen, int signalLen)
    {
        var result = new List<(double[] meaning, double[] signal)>();

        foreach (var line in File.ReadLines(path))
        {
            var parts = line.Split(',').Select(double.Parse).ToArray();

            var meaning = parts[..meaningLen];
            var signal = parts[meaningLen..(meaningLen + signalLen)];

            result.Add((meaning, signal));
        }

        return result;
    }
}
